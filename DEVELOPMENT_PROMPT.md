### 2. image_processor.py - –û–±—Ä–∞–±–æ—Ç–∫–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π

```python
import os
import logging
from typing import List, Tuple, Optional, Union
from PIL import Image, ImageTk, ImageDraw, ImageEnhance
from pathlib import Path
import requests
import zipfile
import tempfile
from config import IMAGE_CONFIG, LOGO_CONFIG

class ImageProcessor:
    """–ö–ª–∞—Å—Å –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –∏ –Ω–∞–Ω–µ—Å–µ–Ω–∏—è –ª–æ–≥–æ—Ç–∏–ø–æ–≤"""
    
    def __init__(self):
        self.supported_formats = IMAGE_CONFIG['supported_formats']
        self.max_preview_size = IMAGE_CONFIG['max_preview_size']
        self.quality = IMAGE_CONFIG['quality']
        
    def load_image(self, file_path: str) -> Optional[Image.Image]:
        """–ó–∞–≥—Ä—É–∂–∞–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ —Å –æ–±—Ä–∞–±–æ—Ç–∫–æ–π –æ—à–∏–±–æ–∫"""
        try:
            if not self.is_supported_format(file_path):
                raise ValueError(f"–ù–µ–ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞–µ–º—ã–π —Ñ–æ—Ä–º–∞—Ç: {file_path}")
            
            image = Image.open(file_path)
            if image.mode not in ('RGB', 'RGBA'):
                image = image.convert('RGB')
            
            return image
        except Exception as e:
            logging.error(f"–û—à–∏–±–∫–∞ –∑–∞–≥—Ä—É–∑–∫–∏ {file_path}: {e}")
            return None
    
    def is_supported_format(self, file_path: str) -> bool:
        """–ü—Ä–æ–≤–µ—Ä—è–µ—Ç –ø–æ–¥–¥–µ—Ä–∂–∫—É —Ñ–æ—Ä–º–∞—Ç–∞ —Ñ–∞–π–ª–∞"""
        return Path(file_path).suffix.lower() in self.supported_formats
    
    def create_preview(self, image: Image.Image, max_size: Tuple[int, int] = None) -> Image.Image:
        """–°–æ–∑–¥–∞–µ—Ç –ø—Ä–µ–≤—å—é –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è"""
        if max_size is None:
            max_size = self.max_preview_size
        
        image_copy = image.copy()
        image_copy.thumbnail(max_size, Image.Resampling.LANCZOS)
        return image_copy
    
    def resize_logo(self, logo: Image.Image, target_image: Image.Image, size_ratio: float) -> Image.Image:
        """–ò–∑–º–µ–Ω—è–µ—Ç —Ä–∞–∑–º–µ—Ä –ª–æ–≥–æ—Ç–∏–ø–∞ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è"""
        target_width = int(target_image.width * size_ratio)
        
        # –°–æ—Ö—Ä–∞–Ω—è–µ–º –ø—Ä–æ–ø–æ—Ä—Ü–∏–∏ –ª–æ–≥–æ—Ç–∏–ø–∞
        aspect_ratio = logo.height / logo.width
        target_height = int(target_width * aspect_ratio)
        
        return logo.resize((target_width, target_height), Image.Resampling.LANCZOS)
    
    def calculate_position(self, image_size: Tuple[int, int], logo_size: Tuple[int, int], 
                         position: str, margin: int = 20) -> Tuple[int, int]:
        """–í—ã—á–∏—Å–ª—è–µ—Ç –ø–æ–∑–∏—Ü–∏—é –ª–æ–≥–æ—Ç–∏–ø–∞ –Ω–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–∏"""
        img_width, img_height = image_size
        logo_width, logo_height = logo_size
        
        position_ratios = LOGO_CONFIG['positions'][position]
        
        # –ë–∞–∑–æ–≤—ã–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã
        x = int((img_width - logo_width) * position_ratios[0])
        y = int((img_height - logo_height) * position_ratios[1])
        
        # –ü—Ä–∏–º–µ–Ω—è–µ–º –æ—Ç—Å—Ç—É–ø—ã
        if position_ratios[0] == 0:  # –õ–µ–≤—ã–π –∫—Ä–∞–π
            x += margin
        elif position_ratios[0] == 1:  # –ü—Ä–∞–≤—ã–π –∫—Ä–∞–π
            x -= margin
        
        if position_ratios[1] == 0:  # –í–µ—Ä—Ö–Ω–∏–π –∫—Ä–∞–π
            y += margin
        elif position_ratios[1] == 1:  # –ù–∏–∂–Ω–∏–π –∫—Ä–∞–π
            y -= margin
        
        return (x, y)
    
    def apply_logo(self, image: Image.Image, logo: Image.Image, 
                   position: Union[str, Tuple[int, int]], 
                   size_ratio: float = None, opacity: float = 1.0, 
                   margin: int = 20) -> Image.Image:
        """–ü—Ä–∏–º–µ–Ω—è–µ—Ç –ª–æ–≥–æ—Ç–∏–ø –∫ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—é"""
        if size_ratio is None:
            size_ratio = LOGO_CONFIG['default_logo_size']
        
        # –°–æ–∑–¥–∞–µ–º –∫–æ–ø–∏—é –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
        result_image = image.copy()
        
        # –ò–∑–º–µ–Ω—è–µ–º —Ä–∞–∑–º–µ—Ä –ª–æ–≥–æ—Ç–∏–ø–∞
        resized_logo = self.resize_logo(logo, image, size_ratio)
        
        # –ü—Ä–∏–º–µ–Ω—è–µ–º –ø—Ä–æ–∑—Ä–∞—á–Ω–æ—Å—Ç—å
        if opacity < 1.0:
            if resized_logo.mode != 'RGBA':
                resized_logo = resized_logo.convert('RGBA')
            
            # –°–æ–∑–¥–∞–µ–º –º–∞—Å–∫—É –ø—Ä–æ–∑—Ä–∞—á–Ω–æ—Å—Ç–∏
            alpha = resized_logo.split()[-1]
            alpha = ImageEnhance.Brightness(alpha).enhance(opacity)
            resized_logo.putalpha(alpha)
        
        # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –ø–æ–∑–∏—Ü–∏—é
        if isinstance(position, str):
            logo_position = self.calculate_position(
                image.size, resized_logo.size, position, margin
            )
        else:
            logo_position = position
        
        # –ù–∞–∫–ª–∞–¥—ã–≤–∞–µ–º –ª–æ–≥–æ—Ç–∏–ø
        if resized_logo.mode == 'RGBA':
            result_image.paste(resized_logo, logo_position, resized_logo)
        else:
            result_image.paste(resized_logo, logo_position)
        
        return result_image
    
    def save_image(self, image: Image.Image, output_path: str, quality: int = None) -> bool:
        """–°–æ—Ö—Ä–∞–Ω—è–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ"""
        try:
            if quality is None:
                quality = self.quality
            
            # –°–æ–∑–¥–∞–µ–º –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—é –µ—Å–ª–∏ –Ω–µ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç
            Path(output_path).parent.mkdir(parents=True, exist_ok=True)
            
            # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –ø–∞—Ä–∞–º–µ—Ç—Ä—ã —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è
            save_kwargs = {}
            if Path(output_path).suffix.lower() in ['.jpg', '.jpeg']:
                save_kwargs['quality'] = quality
                save_kwargs['optimize'] = True
                # –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ–º –≤ RGB –¥–ª—è JPEG
                if image.mode == 'RGBA':
                    background = Image.new('RGB', image.size, (255, 255, 255))
                    background.paste(image, mask=image.split()[-1])
                    image = background
            
            image.save(output_path, **save_kwargs)
            return True
            
        except Exception as e:
            logging.error(f"–û—à–∏–±–∫–∞ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è {output_path}: {e}")
            return False
    
    def download_image(self, url: str, output_path: str) -> bool:
        """–°–∫–∞—á–∏–≤–∞–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –ø–æ URL"""
        try:
            response = requests.get(url, stream=True, timeout=30)
            response.raise_for_status()
            
            with open(output_path, 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192):
                    f.write(chunk)
            
            return True
            
        except Exception as e:
            logging.error(f"–û—à–∏–±–∫–∞ —Å–∫–∞—á–∏–≤–∞–Ω–∏—è {url}: {e}")
            return False
    
    def extract_images_from_zip(self, zip_path: str, extract_to: str) -> List[str]:
        """–ò–∑–≤–ª–µ–∫–∞–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –∏–∑ ZIP –∞—Ä—Ö–∏–≤–∞"""
        extracted_files = []
        
        try:
            with zipfile.ZipFile(zip_path, 'r') as zip_ref:
                for file_info in zip_ref.filelist:
                    if self.is_supported_format(file_info.filename):
                        extracted_path = zip_ref.extract(file_info, extract_to)
                        extracted_files.append(extracted_path)
            
        except Exception as e:
            logging.error(f"–û—à–∏–±–∫–∞ –∏–∑–≤–ª–µ—á–µ–Ω–∏—è –∏–∑ –∞—Ä—Ö–∏–≤–∞ {zip_path}: {e}")
        
        return extracted_files
```

    def _download_images_from_urls(self, urls: List[str]):
        """–°–∫–∞—á–∏–≤–∞–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –ø–æ —Å–ø–∏—Å–∫—É URL"""
        def download_worker():
            download_dir = TEMP_DIR / "downloads"
            download_dir.mkdir(exist_ok=True)
            
            progress_dialog = ProgressDialog(self.root, "–°–∫–∞—á–∏–≤–∞–Ω–∏–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π")
            downloaded_files = []
            
            try:
                for i, url in enumerate(urls):
                    if progress_dialog.cancelled:
                        break
                    
                    progress_dialog.update_progress(
                        int((i / len(urls)) * 100),
                        f"–°–∫–∞—á–∏–≤–∞–Ω–∏–µ {i+1} –∏–∑ {len(urls)}..."
                    )
                    
                    try:
                        # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –∏–º—è —Ñ–∞–π–ª–∞
                        filename = f"image_{i+1}.jpg"
                        if '.' in url.split('/')[-1]:
                            filename = url.split('/')[-1]
                        
                        file_path = download_dir / filename
                        
                        if self.image_processor.download_image(url, str(file_path)):
                            downloaded_files.append(str(file_path))
                        
                    except Exception as e:
                        logger.error(f"–û—à–∏–±–∫–∞ —Å–∫–∞—á–∏–≤–∞–Ω–∏—è {url}: {e}")
                
                progress_dialog.update_progress(100, "–ó–∞–≤–µ—Ä—à–µ–Ω–æ")
                
                if downloaded_files and not progress_dialog.cancelled:
                    self.root.after(0, lambda: self._on_download_complete(downloaded_files))
                
            finally:
                self.root.after(0, progress_dialog.destroy)
        
        threading.Thread(target=download_worker, daemon=True).start()
    
    def _on_download_complete(self, downloaded_files: List[str]):
        """–û–±—Ä–∞–±–æ—Ç–∫–∞ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è —Å–∫–∞—á–∏–≤–∞–Ω–∏—è"""
        self.image_paths = downloaded_files
        self.current_index = 0
        self.processed_images = [None] * len(self.image_paths)
        self._update_ui_state()
        self._load_current_image()
        
        self.status_bar.config(text=f"–°–∫–∞—á–∞–Ω–æ {len(downloaded_files)} –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π")
        logger.info(f"–°–∫–∞—á–∞–Ω–æ {len(downloaded_files)} –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π")

    def _load_current_image(self):
        """–ó–∞–≥—Ä—É–∂–∞–µ—Ç —Ç–µ–∫—É—â–µ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ"""
        if not self.image_paths:
            return
        
        current_path = self.image_paths[self.current_index]
        image = self.image_processor.load_image(current_path)
        
        if image:
            # –ü–æ–∫–∞–∑—ã–≤–∞–µ–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –≤ –ø—Ä–æ—Å–º–æ—Ç—Ä—â–∏–∫–µ

            self.nav_label.config(
                text=f"{self.current_index + 1} –∏–∑ {len(self.image_paths)} - {Path(current_path).name}"
            )
            
            self.current_info.config(
                text=f"–†–∞–∑–º–µ—Ä: {image.width}x{image.height}"
            )

            messagebox.showerror("–û—à–∏–±–∫–∞", f"–ù–µ —É–¥–∞–ª–æ—Å—å –∑–∞–≥—Ä—É–∑–∏—Ç—å –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ: {current_path}")
    
    def _update_preview(self):
        """–û–±–Ω–æ–≤–ª—è–µ—Ç –ø—Ä–µ–¥–≤–∞—Ä–∏—Ç–µ–ª—å–Ω—ã–π –ø—Ä–æ—Å–º–æ—Ç—Ä —Å –ª–æ–≥–æ—Ç–∏–ø–æ–º"""

            return

        image = self.image_processor.load_image(current_path)
        
        if image:
            settings = self.settings_panel.get_settings()
            
            # –ü—Ä–∏–º–µ–Ω—è–µ–º –ª–æ–≥–æ—Ç–∏–ø –¥–ª—è –ø—Ä–µ–¥–≤–∞—Ä–∏—Ç–µ–ª—å–Ω–æ–≥–æ –ø—Ä–æ—Å–º–æ—Ç—Ä–∞
            preview = self.image_processor.apply_logo(
                image,
                self.logo,
                settings['position'],
                settings['size_ratio'],
                settings['opacity'],
                settings['margin']
            )
            
            self.preview_image = preview
            
            # –û–±–Ω–æ–≤–ª—è–µ–º –ø—Ä–æ—Å–º–æ—Ç—Ä—â–∏–∫
